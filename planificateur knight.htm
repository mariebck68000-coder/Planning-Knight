<script type="module">

import {

initializeApp

} from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";

import {

getAuth,

signInAnonymously,

signInWithCustomToken,

onAuthStateChanged

} from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";

import {

getFirestore,

doc,

getDoc,

addDoc,

setDoc,

updateDoc,

deleteDoc,

onSnapshot,

collection,

query,

where,

getDocs

} from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

 

// Firebase Global Variables (Provided by Canvas)

const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';

const firebaseConfig = JSON.parse(typeof __firebase_config !== 'undefined' ? __firebase_config : '{}');

const initialAuthToken = typeof __initial_auth_token !== 'undefined' ? __initial_auth_token : null;

 

// --- Firebase Initialization and Auth ---

let app, db, auth, userId;

const gmData = {

dates: [],

scenarios: [],

availabilities: {}

};

 

const initFirebase = async () => {

try {

if (Object.keys(firebaseConfig).length > 0) {

app = initializeApp(firebaseConfig);

db = getFirestore(app);

auth = getAuth(app);

console.log('Firebase initialized.');

 

if (initialAuthToken) {

await signInWithCustomToken(auth, initialAuthToken);

console.log('Signed in with custom token.');

} else {

await signInAnonymously(auth);

console.log('Signed in anonymously.');

}

 

onAuthStateChanged(auth, (user) => {

if (user) {

userId = user.uid;

document.getElementById('loading').classList.add('hidden');

console.log('User authenticated:', userId);

setupUI();

loadData();

} else {

console.log('No user signed in.');

}

});

} else {

console.error('Firebase config is not available. Please ensure the app is deployed correctly.');

}

} catch (error) {

console.error("Firebase initialization failed:", error);

}

};

 

const setupUI = () => {

const mgmtPassword = "knight"; // Simple password for GM view

const mgmtLoginBtn = document.getElementById('mgmt-login-btn');

const passwordInput = document.getElementById('mgmt-password');

const mainContainer = document.getElementById('main-container');

 

mainContainer.classList.remove('hidden');

 

mgmtLoginBtn.addEventListener('click', () => {

if (passwordInput.value === mgmtPassword) {

document.getElementById('player-view').classList.add('hidden');

document.getElementById('mgmt-view').classList.remove('hidden');

document.getElementById('login-section').classList.add('hidden');

// Store GM state in a way that is not cleared on reload

sessionStorage.setItem('isGM', 'true');

} else {

showMessage('Mot de passe incorrect.', 'error');

}

});

 

if (sessionStorage.getItem('isGM') === 'true') {

document.getElementById('player-view').classList.add('hidden');

document.getElementById('mgmt-view').classList.remove('hidden');

document.getElementById('login-section').classList.add('hidden');

} else {

document.getElementById('player-view').classList.remove('hidden');

document.getElementById('mgmt-view').classList.add('hidden');

}

};

 

const resetUI = () => {

document.getElementById('login-section').classList.remove('hidden');

document.getElementById('player-view').classList.add('hidden');

document.getElementById('mgmt-view').classList.add('hidden');

document.getElementById('mgmt-password').value = '';

sessionStorage.removeItem('isGM');

};

 

const showMessage = (msg, type = 'info') => {

const msgBox = document.getElementById('message-box');

msgBox.textContent = msg;

msgBox.className = 'p-3 my-4 rounded-md text-white text-center';

if (type === 'error') {

msgBox.classList.add('bg-red-500');

} else if (type === 'success') {

msgBox.classList.add('bg-green-500');

} else {

msgBox.classList.add('bg-blue-500');

}

msgBox.classList.remove('hidden');

setTimeout(() => msgBox.classList.add('hidden'), 5000);

};

 

// --- Firestore Data Operations ---

const getPublicCollectionPath = (collectionName) => {

return `/artifacts/${appId}/public/data/${collectionName}`;

};

 

const getPlayerCollectionPath = (collectionName) => {

return `/artifacts/${appId}/users/${userId}/${collectionName}`;

};

 

const addDate = async (date) => {

try {

await addDoc(collection(db, getPublicCollectionPath('dates')), {

date: date,

timestamp: new Date(date).getTime()

});

showMessage('Date ajoute avec succs !');

} catch (e) {

console.error("Erreur d'ajout de date: ", e);

showMessage("Erreur lors de l'ajout de la date.", 'error');

}

};

 

const addScenario = async (scenario) => {

try {

await addDoc(collection(db, getPlayerCollectionPath('scenarios')), scenario);

showMessage('Scnario ajout avec succs !');

} catch (e) {

console.error("Erreur d'ajout de scnario: ", e);

showMessage("Erreur lors de l'ajout du scnario.", 'error');

}

};

 

const updateAvailability = async (playerId, dateId, isAvailable) => {

try {

const docRef = doc(db, getPublicCollectionPath('availabilities'), `${playerId}_${dateId}`);

await setDoc(docRef, {

playerId: playerId,

dateId: dateId,

available: isAvailable

});

} catch (e) {

console.error("Erreur de mise jour de la disponibilit: ", e);

}

};

 

// --- UI Listeners and Logic ---

const loadData = () => {

if (sessionStorage.getItem('isGM') === 'true') {

const datesRef = collection(db, getPublicCollectionPath('dates'));

const scenariosRef = collection(db, getPlayerCollectionPath('scenarios'));

const availabilitiesRef = collection(db, getPublicCollectionPath('availabilities'));

 

onSnapshot(datesRef, (snapshot) => {

gmData.dates = snapshot.docs.map(d => ({ ...d.data(), id: d.id })).sort((a, b) => a.timestamp - b.timestamp);

updateGMUI();

});

 

onSnapshot(scenariosRef, (snapshot) => {

gmData.scenarios = snapshot.docs.map(d => ({ ...d.data(), id: d.id }));

updateGMUI();

});

 

onSnapshot(availabilitiesRef, (snapshot) => {

gmData.availabilities = {};

snapshot.forEach(doc => {

const data = doc.data();

if (!gmData.availabilities[data.playerId]) {

gmData.availabilities[data.playerId] = {};

}

gmData.availabilities[data.playerId][data.dateId] = data.available;

});

updateGMUI();

});

}

 

const datesRef = collection(db, getPublicCollectionPath('dates'));

onSnapshot(datesRef, (snapshot) => {

const dates = snapshot.docs.map(d => ({ ...d.data(), id: d.id })).sort((a, b) => a.timestamp - b.timestamp);

displayPlayerView(dates);

});

};

 

const updateGMUI = () => {

displayGMView(gmData.dates, gmData.availabilities);

displayScenarios(gmData.scenarios);

calculateMatches(gmData.dates, gmData.scenarios, gmData.availabilities);

};

 

const displayGMView = (dates, availabilities) => {

const availabilitiesTable = document.getElementById('availabilities-table');

availabilitiesTable.innerHTML = '';

const playerNames = [...new Set(Object.keys(availabilities))];

 

// Header Row

let headerHtml = '<tr><th class="p-2 border-b-2">Joueur</th>';

dates.forEach(date => {

headerHtml += `<th class="p-2 border-b-2">${new Date(date.date).toLocaleDateString('fr-FR', { month: 'short', day: 'numeric', hour: '2-digit', minute: '2-digit' })}</th>`;

});

headerHtml += '</tr>';

availabilitiesTable.innerHTML += headerHtml;

 

// Data Rows

playerNames.forEach(playerId => {

let rowHtml = `<tr><td class="p-2 border-b-2">${playerId}</td>`;

dates.forEach(date => {

const isAvailable = availabilities[playerId] && availabilities[playerId][date.id] === true;

const availabilityText = isAvailable ? 'Oui' : 'Non';

const availabilityColor = isAvailable ? 'text-green-600' : 'text-red-600';

rowHtml += `<td class="p-2 border-b-2 ${availabilityColor}">${availabilityText}</td>`;

});

rowHtml += '</tr>';

availabilitiesTable.innerHTML += rowHtml;

});

};

 

const displayScenarios = (scenarios) => {

const scenarioList = document.getElementById('scenario-list');

scenarioList.innerHTML = '';

if (scenarios.length === 0) {

scenarioList.innerHTML = '<p class="text-center text-gray-500">Aucun scnario ajout pour le moment.</p>';

return;

}

 

scenarios.forEach(scenario => {

const scenarioHtml = `

<div class="bg-white p-4 rounded-lg shadow-sm">

<h4 class="font-bold text-lg text-gray-800">${scenario.name}</h4>

<p class="text-gray-600">Parties requises: ${scenario.requiredParts}</p>

<p class="text-gray-600">Joueurs requis: ${scenario.requiredPlayers}</p>

<p class="text-gray-600">Joueurs ligibles: ${scenario.eligiblePlayers}</p>

</div>

`;

scenarioList.innerHTML += scenarioHtml;

});

};

 

const displayPlayerView = async (dates) => {

const datesList = document.getElementById('dates-list');

datesList.innerHTML = '';

if (dates.length === 0) {

datesList.innerHTML = '<p class="text-center text-gray-500">Aucune date propose pour le moment.</p>';

return;

}

 

const currentPlayerName = localStorage.getItem('playerName') || '';

const playerNameInput = document.getElementById('player-name');

playerNameInput.value = currentPlayerName;

 

const availabilitiesRef = collection(db, getPublicCollectionPath('availabilities'));

const playerAvailabilities = {};

const availabilitiesSnapshot = await getDocs(availabilitiesRef);

availabilitiesSnapshot.forEach(doc => {

const data = doc.data();

if (data.playerId === currentPlayerName) {

playerAvailabilities[data.dateId] = data.available;

}

});

 

dates.forEach(date => {

const isAvailable = playerAvailabilities[date.id] === true;

const dateElement = document.createElement('div');

dateElement.className = 'flex items-center justify-between p-3 my-2 bg-gray-100 rounded-lg shadow-sm';

dateElement.innerHTML = `

<span class="font-semibold text-gray-700">${new Date(date.date).toLocaleString('fr-FR', { dateStyle: 'full', timeStyle: 'short' })}</span>

<label class="flex items-center space-x-2 cursor-pointer">

<span class="text-gray-600">Disponible ?</span>

<input type="checkbox" class="toggle toggle-primary" ${isAvailable ? 'checked' : ''} data-date-id="${date.id}">

</label>

`;

datesList.appendChild(dateElement);

});

};

 

const savePlayerAvailability = async () => {

const playerName = document.getElementById('player-name').value.trim();

if (!playerName) {

showMessage("Veuillez entrer votre nom d'abord.", 'error');

return;

}

localStorage.setItem('playerName', playerName);

 

const checkboxes = document.querySelectorAll('#dates-list input[type="checkbox"]');

for (const checkbox of checkboxes) {

const dateId = checkbox.getAttribute('data-date-id');

const isAvailable = checkbox.checked;

await updateAvailability(playerName, dateId, isAvailable);

}

showMessage('Disponibilits enregistres avec succs !', 'success');

};

 

// --- Main Calculation Logic ---

const calculateMatches = (dates, scenarios, availabilities) => {

const matchesDiv = document.getElementById('matches-container');

matchesDiv.innerHTML = '<p class="text-center text-gray-500 my-4">Calcul des correspondances en cours...</p>';

 

const matches = [];

 

scenarios.forEach(scenario => {

const eligiblePlayers = scenario.eligiblePlayers.split(',').map(p => p.trim());

const requiredPlayers = parseInt(scenario.requiredPlayers, 10);

const requiredParts = parseInt(scenario.requiredParts, 10);

 

if (eligiblePlayers.length < requiredPlayers) return;

 

const playerCombinations = getCombinations(eligiblePlayers, requiredPlayers);

 

playerCombinations.forEach(group => {

const groupAvailabilities = dates.filter(date => {

return group.every(player => availabilities[player] && availabilities[player][date.id] === true);

});

 

if (groupAvailabilities.length >= requiredParts) {

const match = {

scenario: scenario.name,

group: group.join(', '),

dates: groupAvailabilities.slice(0, requiredParts).map(d => new Date(d.date).toLocaleString('fr-FR')),

excessDates: groupAvailabilities.length - requiredParts

};

matches.push(match);

}

});

});

 

displayMatches(matches);

};

 

const displayMatches = (matches) => {

const matchesDiv = document.getElementById('matches-container');

if (matches.length === 0) {

matchesDiv.innerHTML = '<p class="text-center text-gray-500 my-4">Aucune correspondance trouve. Moins de joueurs disponibles ou de dates proposes.</p>';

return;

}

 

matchesDiv.innerHTML = '';

matches.forEach((match, index) => {

const matchHtml = `

<div class="bg-indigo-50 p-4 rounded-lg shadow-md my-4">

<h4 class="font-bold text-lg text-indigo-800 mb-2">Correspondance ${index + 1}</h4>

<p><strong>Scnario:</strong> ${match.scenario}</p>

<p><strong>Joueurs:</strong> ${match.group}</p>

<p><strong>Dates proposes:</strong> ${match.dates.join(', ')}</p>

${match.excessDates > 0 ? `<p class="text-green-600">Il reste ${match.excessDates} dates supplmentaires pour ce groupe !</p>` : ''}

</div>

`;

matchesDiv.innerHTML += matchHtml;

});

};

 

function getCombinations(array, size) {

const result = [];

function generate(current, start) {

if (current.length === size) {

result.push([...current]);

return;

}

for (let i = start; i < array.length; i++) {

current.push(array[i]);

generate(current, i + 1);

current.pop();

}

}

generate([], 0);

return result;

}

 

// --- Event Listeners for Forms ---

window.onload = initFirebase;

window.calculateMatches = calculateMatches; // Expose to HTML

window.resetUI = resetUI; // Expose to HTML

window.savePlayerAvailability = savePlayerAvailability; // Expose to HTML

 

const dateForm = document.getElementById('add-date-form');

dateForm.addEventListener('submit', (e) => {

e.preventDefault();

const dateInput = document.getElementById('new-date');

if (dateInput.value) {

addDate(dateInput.value);

dateInput.value = '';

}

});

 

const scenarioForm = document.getElementById('add-scenario-form');

scenarioForm.addEventListener('submit', (e) => {

e.preventDefault();

const scenario = {

name: document.getElementById('scenario-name').value,

requiredParts: document.getElementById('scenario-parts').value,

requiredPlayers: document.getElementById('scenario-players').value,

eligiblePlayers: document.getElementById('scenario-eligible').value

};

addScenario(scenario);

scenarioForm.reset();

});

</script>
